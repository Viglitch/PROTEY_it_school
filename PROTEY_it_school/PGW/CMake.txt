cmake_minimum_required(VERSION 3.14)
project(pgw_server CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

enable_testing()

include(FetchContent)

FetchContent_Declare(
  json
  GIT_REPOSITORY https://github.com/nlohmann/json.git
  GIT_TAG v3.11.2
)
FetchContent_MakeAvailable(json)

FetchContent_Declare(
  googletest
  GIT_REPOSITORY https://github.com/google/googletest.git
  GIT_TAG release-1.12.1
)
FetchContent_MakeAvailable(googletest)

add_executable(pgw_server
  src/main.cpp
  src/pgw_server.cpp
  src/pgw_client.cpp
  include/pgw_server.h
)

target_include_directories(pgw_server PRIVATE include)
target_link_libraries(pgw_server PRIVATE pthread)

add_executable(pgw_tests
  tests/pgw_server_test.cpp
  src/pgw_server.cpp
  src/pgw_client.cpp
  include/pgw_server.h
)

target_include_directories(pgw_tests PRIVATE include)
target_link_libraries(pgw_tests PRIVATE
  gtest_main
  pthread
)

add_test(NAME pgw_tests COMMAND pgw_tests)

install(TARGETS pgw_server DESTINATION bin)
